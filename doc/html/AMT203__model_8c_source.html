38 static void amt203_xfer_byte(amt203_t * amt203, uint8_t sbyte, uint8_t * rbyte)
39 {
40  spi_t * spi = amt203->spi;
41 
42  /* Every transfer must be gated by the chip select signal. */
43  amt203->chip_select(1);
44  spi_xfer(spi, sbyte, rbyte);
45  amt203->chip_select(0);

47  /* Be safe */
48  amt203_delay();

51 static int amt203_wait_for_ack(amt203_t * amt203, uint8_t cmd, int max_errors)
52 {
53  int errors = 0;
54  uint8_t res = 0;
55  while (res != cmd) {
56  amt203_xfer_byte(amt203, cmd, &res);
57  if (res != AMT203_RES_WAIT) {
58  errors++;
59  if (errors > max_errors) { return -1; }
60  }
61  }
62  return 0;
63 }

81 int amt203_position_read(amt203_t * amt203, uint16_t * position)
82 {
83  uint8_t cmd = AMT203_CMD_RD_POS;
84  uint8_t res = 0;
85  assert(amt203 != NULL);
86 
87  /* Send command */
88  amt203_xfer_byte(amt203, cmd, &res);
89  if (amt203_wait_for_ack(amt203, cmd, 5) != 0) { return -1; }
90 
91  /* Get high byte */
92  amt203_xfer_byte(amt203, AMT203_CMD_NOP, &res);
93  *position |= ((uint16_t)res << 8);
94 
95  /* Get low byte */
96  amt203_xfer_byte(amt203, AMT203_CMD_NOP, &res);
97  *position |= (uint16_t)res;
98 
99  return (*position > 0x0fff);
100 }
