00632 void TextOutRotated(CDC* pDC, const CString str, int x, int y, double angle)
00633 {
00634    // convert angle to radian
00635    double pi = 3.141592654;
00636    double radian = pi * 2 / 360 * angle;
00637  
00638    // get the center of a not-rotated text
00639    CSize TextSize = pDC->GetTextExtent(str);
00640    CPoint center;
00641    center.x = TextSize.cx / 2;
00642    center.y = TextSize.cy / 2;
00643  
00644    // now calculate the center of the rotated text
00645    CPoint rcenter;
00646    rcenter.x = long(cos(radian) * center.x - sin(radian) * center.y);
00647    rcenter.y = long(sin(radian) * center.x + cos(radian) * center.y);
00648  
00649    // finally draw the text and move it to the center of the rectangle
00650    pDC->SetTextAlign(TA_BASELINE);
00651    pDC->SetBkMode(TRANSPARENT);
00652    pDC->TextOut(x - rcenter.x, y + rcenter.y, str);
